SYSTEM INFO
+ Check system is latest version or not?
- Expected outcome: No new software available
```
softwareupdate --list
```

+ List out all applications that does not come by defaults
```
ls /Applications | grep -v -e "Preview.app" -e "Safari.app" -e "Mail.app" -e "Calendar.app" -e "Messages.app" -e "Contacts.app" -e "Photos.app" -e "FaceTime.app" -e "Reminders.app" -e "Notes.app" -e "Music.app" -e "TV.app" -e "Podcasts.app" -e "News.app" -e "Stocks.app" -e "VoiceMemos.app"

mdfind "kMDItemContentType == 'com.apple.application-bundle' && kMDItemAppStoreHasReceipt != 1" -onlyin /Applications | while read -r app; do
  version=$(mdls -name kMDItemVersion "$app" | awk -F'"' '{print $2}')
  echo "$app - Version: $version"
done

ls /Applications | while read -r app; do
  if [[ -d "/Applications/$app/Contents" ]]; then
    version=$(defaults read "/Applications/$app/Contents/Info.plist" CFBundleShortVersionString 2>/dev/null)
    echo "$app - Version: ${version:-Unknown}"
  fi
done

```
